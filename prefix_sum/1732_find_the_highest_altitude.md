```php
class Solution {

    /**
     * @param Integer[] $gain
     * @return Integer
     */
    function largestAltitude($gain) {
        $highest = 0;
        $altitude = 0;

        foreach ($gain as $alt) {
            $altitude += $alt;
            $highest = max($highest, $altitude);
        }

        return $highest;
    }
}
```
